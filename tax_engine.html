async function generateTaxEngine(){ const selectedPeriod = document.getElementById("periodeSelect").value; // Ambil payroll sampai periode terpilih (untuk akumulasi tahunan) const { data: payrollData, error: payrollError } = await supabaseClient .from('payroll') .select('*') .lte('periode', selectedPeriod) .order('nik') .order('periode'); if(payrollError){ console.error(payrollError); alert("Gagal ambil data payroll"); return; } const { data: masterData, error: masterError } = await supabaseClient .from('master_payroll') .select('nik,nama,project,status_pajak'); if(masterError){ console.error(masterError); alert("Gagal ambil master payroll"); return; } const masterMap = {}; masterData.forEach(m => { masterMap[m.nik] = m; }); // Kelompokkan per pegawai per tahun const grouped = {}; payrollData.forEach(row => { const tahun = row.periode.split('-')[0]; const key = `${row.nik}_${tahun}`; if(!grouped[key]) grouped[key] = []; grouped[key].push(row); }); const results = []; for(const key in grouped){ let akBruto = 0; let akIuran = 0; let akTER = 0; const rows = grouped[key]; const nik = rows[0].nik; const master = masterMap[nik] || {}; const status = normalizeStatus(master.status_pajak); const ptkp = PTKP_MAP[status] || 54000000; for(const row of rows){ const gapok = Number(row.gaji_pokok || 0); const tunjangan = Number(row.tunjangan || 0); const premi = Number(row.jkk || 0) + Number(row.jkm || 0) + Number(row.bpjs4 || 0); const irregular = Number(row.irregular || 0); const iuran = Number(row.jht2 || 0) + Number(row.jp1 || 0); const bruto = gapok + tunjangan + premi + irregular; akBruto += bruto; akIuran += iuran; const biayaJabatan = Math.min(akBruto * 0.05, 6000000); const neto = akBruto - biayaJabatan - akIuran; const pkp = Math.max(0, neto - ptkp); // Hitung TER bulan berjalan const terRate = getTERPercent(bruto, status); const pphTer = Math.round(bruto * terRate); akTER += pphTer; // Hitung PPh final progresif const pphFinal = getProgressivePPh21Annual(pkp); const selisih = pphFinal - akTER; let ket = "Nihil"; if(selisih > 0) ket = "Kurang Potong"; if(selisih < 0) ket = "Lebih Potong"; // Simpan hanya periode terpilih if(row.periode === selectedPeriod){ results.push({ periode: row.periode, nik: row.nik, nama: master.nama || '', project: master.project || '', gapok, tunjangan, premi_asuransi: premi, irregular, bruto_bulan: bruto, akumulasi_bruto: akBruto, akumulasi_biaya_jabatan: biayaJabatan, iuran_pensiun_jht: akIuran, jumlah_neto: neto, ptkp, pkp, pph_ter: pphTer, pph_21_final: pphFinal, selisih_pajak: selisih, keterangan: ket }); } } } const { error: upsertError } = await supabaseClient .from('tax_engine') .upsert(results, { onConflict: 'nik,periode' }); if(upsertError){ console.error(upsertError); alert("Gagal simpan tax engine"); return; } alert("Generate Tax berhasil untuk periode " + selectedPeriod); loadTaxEngine(); }
