async function loadPPH(){ const tahun=document.getElementById("tahun").value; const filterNik=document.getElementById("filterNik").value.toLowerCase(); const filterProject=document.getElementById("filterProject").value.toLowerCase(); // ================= LOAD DATA ================= const {data:payroll=[]}=await supabaseClient .from("payroll") .select("*") .like("periode",`${tahun}-%`); const {data:tax=[]}=await supabaseClient .from("tax_engine") .select("*") .like("periode",`${tahun}-%`); const {data:master=[]}=await supabaseClient .from("master_payroll") .select("*"); // ================= MAP ================= let map={}; // 🔥 payroll (akumulasi) payroll.forEach(r=>{ if(!map[r.nik]){ map[r.nik]={ gapok:0, tunjangan:0, irregular:0, bulanTerakhir:"" }; } map[r.nik].gapok += +r.gaji_pokok || 0; map[r.nik].tunjangan += +r.tunjangan || 0; map[r.nik].irregular += +r.irregular || 0; if(r.periode > map[r.nik].bulanTerakhir){ map[r.nik].bulanTerakhir = r.periode; } }); // 🔥 tax_engine (pakai nilai jadi, JANGAN hitung ulang) tax.forEach(t=>{ if(!map[t.nik]) return; // premi → SUM map[t.nik].premi = (map[t.nik].premi || 0) + (+t.premi || 0); // TER → SUM map[t.nik].ter = (map[t.nik].ter || 0) + (+t.pph_ter || 0); // ambil nilai terbesar (akumulasi terakhir) map[t.nik].bruto = Math.max(map[t.nik].bruto || 0, +t.akumbruto || 0); map[t.nik].jht = Math.max(map[t.nik].jht || 0, +t.iuran_pensiun || 0); // 🔥 ambil BULAN TERAKHIR if(t.periode === map[t.nik].bulanTerakhir){ map[t.nik].bj = +t.biaya_jabatan || 0; map[t.nik].neto = +t.neto || 0; map[t.nik].pkp = +t.pkp || 0; map[t.nik].final = +t.pph21_final || 0; } }); // ================= BUILD ================= pphData=[]; Object.keys(map).forEach(nik=>{ let m = master.find(x=>x.nik===nik) || {}; let d = map[nik]; // FILTER if(filterNik && !nik.toLowerCase().includes(filterNik)) return; if(filterProject && !(m.project||"").toLowerCase().includes(filterProject)) return; let pengurang = (d.bj || 0) + (d.jht || 0); pphData.push({ nik, nama:m.nama||"", project:m.project||"", gapok:d.gapok||0, tunjangan:d.tunjangan||0, premi:d.premi||0, irregular:d.irregular||0, bruto:d.bruto||0, biaya_jabatan:d.bj||0, jht:d.jht||0, pengurang, neto:d.neto||0, // ✅ dari DB ptkp:getPTKP(m.status_pajak), pkp:d.pkp||0, // ✅ dari DB pph_final:d.final||0, pph_potong:d.ter||0, selisih:(d.final||0)-(d.ter||0), status:Math.abs((d.final||0)-(d.ter||0))<1000?"VALID":"CEK" }); }); render(); renderChart(); }
